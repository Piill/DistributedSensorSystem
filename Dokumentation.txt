init:
	Load sensor type definitions
	Connect to wifi


main loop:
	if no wifi:
		connect to wifi
	
	connect to server
	load config from server
	check sensors
	report sensor values to server
	close tcp connection
	sleep


Protokol:
Message type (Byte {Register = 0x1, Config =0x2, Data=0x3, Error = 0x4, Ok = 0x5})
Data (Byte array)

Error:
(No data)

Register:
DeviceID (int)

Config:
NumSensors  (int)
{
	SensorID (int)
	SensorType (int)
	SensorPin (int)
} repeat

Data:
NumSensors (int)
{
	SensorID (int)
	SensorData (int)
} repeat
